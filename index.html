const express = require('express'); const { Aki } = require('aki-api'); const app = express(); app.use(express.json()); // Храним сессии Акинатора в памяти (ключ — sessionId) const sessions = new Map(); // ================== API ================== // Запуск новой игры app.post('/api/start', async (req, res) => { try { const region = req.body.region || 'ru'; const aki = new Aki({ region }); await aki.start(); const sessionId = Date.now().toString(36) + Math.random().toString(36).substr(2); sessions.set(sessionId, aki); res.json({ sessionId, question: aki.question, progress: aki.progress, step: aki.currentStep }); } catch (err) { res.status(500).json({ error: err.message }); } }); // Ответ на вопрос (0=Да, 1=Нет, 2=Не знаю, 3=Возможно, 4=Возможно нет) app.post('/api/answer', async (req, res) => { try { const { sessionId, answer } = req.body; const aki = sessions.get(sessionId); if (!aki) return res.status(404).json({ error: 'Сессия не найдена' }); await aki.step(answer); if (aki.progress >= 90 || aki.currentStep >= 78) { await aki.win(); const guesses = aki.answers; sessions.delete(sessionId); return res.json({ finished: true, guesses }); } res.json({ finished: false, question: aki.question, progress: aki.progress, step: aki.currentStep }); } catch (err) { res.status(500).json({ error: err.message }); } }); // ================== ВСТРОЕННАЯ СТРАНИЦА ================== const html = ` Акинатор · Тёмная версия
`; // Отдаём HTML для любого GET-запроса (кроме API) app.get('*', (req, res) => { res.send(html); }); // ================== ЗАПУСК ================== const PORT = process.env.PORT || 3000; app.listen(PORT, () => { console.log(`Акинатор запущен на http://localhost:${PORT}`); });
